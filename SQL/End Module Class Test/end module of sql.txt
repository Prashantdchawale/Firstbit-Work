q16_____________
select c.region, sum(to.total_amount) as total_orders
from orders to
join customers c 
on to.customer_id = c.customer_id
group by c.region
having sum(to.total_amount)>10000;





q17____________

select b.author_id, sum(s.quantity) as total_books_sold
from books b
join sales s on b.book_id=s.book_id
group by b.author_id
having sum(s.quantity)>100;
